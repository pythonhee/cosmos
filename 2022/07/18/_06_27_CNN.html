def cost(x, y, w):
  c = 0
  for i in range(len(x)):
    hx = x[i] * w 
    c += (hx - y[i]) ** 2
  
  return c / len(x)

def gradient_descent(x, y, w):
  c = 0
  for i in range(len(x)):
    hx = w * x[i]
    # c += [2 * (hx - y[i])] * (hx - y[i])
    # c += [2 * (hx - y[i])] * (w * x[i] - y[i])
    # c += 2 * hx - y[i] * x[i]
    c += (hx - y[i]) * x[i]

  return c / len(x)
