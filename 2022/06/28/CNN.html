import pandas as pd
import tensorflow as tf

def make_xy():
  cars = pd.read_csv('/content/cars.csv', index_col=0)

  x = cars.values[:, 0]
  y = cars.values[:, 1]

  return x, y

def linear_regression_cars():
  def predict(x, w, b):
    return w * x + b

  def mean_square_error(y, p):
